# prompt: 使用go 实现一个鸡尾酒排序
# answer: 
 ```go
package main

import (
	"fmt"
	"sort"
)

// Cocktail 代表一个鸡尾酒
type Cocktail struct {
	Name     string
	Ingredients []string // 鸡尾酒的配料列表
	Difficulty int       // 难度等级 (1-5, 1为简单，5为困难)
}

// 比较函数，用于对 Cocktail 进行排序
// 首先按难度排序，然后按名称排序
func (c *Cocktail) Len() int {
	return len(c.Ingredients) // 使用配料数量作为长度，方便排序
}

func (c *Cocktail) Swap(i, j int) {
	c.Ingredients[i], c.Ingredients[j] = c.Ingredients[j], c.Ingredients[i]
}

func (c *Cocktail) Less(i, j int) bool {
	// 首先比较难度，难度越低越靠前
	if c.Difficulty != j.Difficulty {
		return c.Difficulty < j.Difficulty
	}
	// 如果难度相同，则按名称排序
	return c.Name < j.Name
}

// CocktailList 是 Cocktail 的切片
type CocktailList []Cocktail

// sortCocktails 使用 sort.Slice 对 CocktailList 进行排序
func (cl CocktailList) Sort() {
	sort.Slice(cl, func(i, j int) bool {
		return cl[i].Less(j)
	})
}

// 示例鸡尾酒列表
func main() {
	cocktails := []Cocktail{
		{Name: "Old Fashioned", Ingredients: []string{"Bourbon", "Sugar", "Angostura Bitters", "Water"}, Difficulty: 2},
		{Name: "Margarita", Ingredients: []string{"Tequila", "Lime Juice", "Orange Liqueur", "Salt"}, Difficulty: 3},
		{Name: "Manhattan", Ingredients: []string{"Rye Whiskey", "Sweet Vermouth", "Bitters"}, Difficulty: 3},
		{Name: "Daiquiri", Ingredients: []string{"Rum", "Lime Juice", "Sugar"}, Difficulty: 1},
		{Name: "Cosmopolitan", Ingredients: []string{"Vodka", "Cranberry Juice", "Lime Juice", "Orange Liqueur"}, Difficulty: 4},
		{Name: "Mojito", Ingredients: []string{"Rum", "Mint", "Lime Juice", "Sugar", "Soda Water"}, Difficulty: 2},
		{Name: "Negroni", Ingredients: []string{"Gin", "Campari", "Sweet Vermouth"}, Difficulty: 3},
	}

	// 对鸡尾酒列表进行排序
	cocktailsList := CocktailList(cocktails)
	cocktailsList.Sort()

	// 打印排序后的鸡尾酒列表
	fmt.Println("排序后的鸡尾酒列表：")
	for _, cocktail := range cocktailsList {
		fmt.Printf("名称: %s, 配料: %v, 难度: %d\n", cocktail.Name, cocktail.Ingredients, cocktail.Difficulty)
	}
}
```

**代码解释:**

1. **`Cocktail` 结构体:**
   - `Name`: 鸡尾酒的名称 (string)。
   - `Ingredients`: 鸡尾酒的配料列表 (string 的切片)。
   - `Difficulty`: 鸡尾酒的难度等级 (int，1-5，1为简单，5为困难)。

2. **`CocktailList` 类型:**
   - `CocktailList` 是 `Cocktail` 结构体的切片，方便进行排序操作。

3. **`Len()`, `Swap()`, `Less()` 方法:**
   - 这些方法实现了 `sort.Interface` 接口，这是 `sort.Slice` 函数所需要的接口。
   - `Len()`: 返回切片的长度。
   - `Swap()`: 交换切片中两个元素的顺序。
   - `Less()`:  定义了排序的比较逻辑。  首先比较难度，难度越低越靠前。如果难度相同，则按名称排序。

4. **`Sort()` 方法:**
   - 使用 `sort.Slice` 函数对 `CocktailList` 进行排序。
   - `sort.Slice` 函数接受一个切片和一个比较函数作为参数。
   - 比较函数 `func(i, j int) bool { return cl[i].Less(j) }` 调用 `Less()` 方法进行比较，并返回结果。

5. **`main()` 函数:**
   - 创建一个 `Cocktail` 结构体的切片，并填充一些示例数据。
   - 将 `Cocktail` 切片转换为 `CocktailList` 类型。
   - 调用 `Sort()` 方法对 `CocktailList` 进行排序。
   - 打印排序后的鸡尾酒列表。
